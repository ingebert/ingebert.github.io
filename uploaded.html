Visualizing CSVs with Pandas, Seaborn, Panel and PyScript

import io import pandas as pd import panel as pn import seaborn as sns from matplotlib.figure import Figure pn.config.sizing_mode="stretch_width" fileInput = pn.widgets.FileInput(accept=".csv").servable(target="fileinput") @pn.depends(fileInput) def to_plot(file: bytes): if file is None: return "Please upload a `.csv` file with columns `arpi` and `total_installs`." data = pd.read_csv(io.BytesIO(file)) x = data['arpi'] #change column name y = data['total_installs'] #change column name fig = Figure(figsize=(10, 6)) ax = fig.add_subplot(111) sns.lineplot(x=x, y=y, data=data, ax=ax) return fig pn.Column( to_plot ).servable(target="op") None # For some reason an error is raised if I end with .servable()
